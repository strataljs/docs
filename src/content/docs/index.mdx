<CardGrid stagger>
	<Card title="Automatic OpenAPI Docs" icon="open-book">

	</Card>
	<Card title="NestJS-style Modules" icon="puzzle">
		Organize your application into cohesive, reusable modules with clear boundaries and encapsulation.
	</Card>
	<Card title="Dependency Injection" icon="setting">
		Built-in DI container for clean, testable architecture without manual wiring.
	</Card>
	<Card title="Convention-based REST" icon="rocket">
